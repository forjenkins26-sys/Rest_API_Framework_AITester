# Page Object Template

Use this as the structural template for any `src/pages/<name>.page.ts` generated by the `create-page-object` skill. Adapt locator strategies and method bodies to the user's described page; preserve the structure.

## Anatomy

```ts
import { Locator, Page, expect } from '@playwright/test';
import { BasePage } from './base.page';
// Uncomment if the page has the global header/footer:
// import { HeaderComponent } from '../components/header.component';
// import { FooterComponent } from '../components/footer.component';
import { ROUTES } from '../utils/constants';

export class <Name>Page extends BasePage {
  // 1) Optional shared components — same instances used across pages
  // readonly header: HeaderComponent;
  // readonly footer: FooterComponent;

  // 2) Locators — declared readonly, named after the user intent, not the DOM
  readonly title: Locator;
  readonly <fieldName>: Locator;
  // ... add one Locator per locator the user provided

  constructor(page: Page) {
    super(page);
    // this.header = new HeaderComponent(page);
    // this.footer = new FooterComponent(page);
    this.title = page.locator('[data-test="title"]');
    this.<fieldName> = page.getByRole('<role>', { name: '<accessible name>' });
    // ... bind one Locator per declared field
  }

  // 3) Navigation
  async goto(): Promise<void> {
    await this.navigate(ROUTES.<routeKey>);
    await this.expectLoaded();
  }

  // 4) Load assertion — every page MUST expose this
  async expectLoaded(): Promise<void> {
    await expect(this.page).toHaveURL(new RegExp(`${ROUTES.<routeKey>}$`));
    await expect(this.title).toHaveText('<expected heading>');
  }

  // 5) Action methods — named for user intent, web-first assertions inside
  async <actionName>(/* args */): Promise<void> {
    await this.<fieldName>.click(); // or fill / selectOption etc.
    // assert post-condition where reasonable
  }
}
```

## Rules

- **Selector priority**: `getByRole` > `getByLabel` > `getByPlaceholder` > `getByText` > `getByTestId` / `[data-test="..."]` > CSS. Drop down the list only when forced.
- **No sleeps**. Replace any urge to `waitForTimeout` with `await expect(locator).toBeVisible()` or another web-first assertion.
- **Locator naming**: name fields for the *user's intent*, not the underlying element (`saveButton` not `btn1`).
- **Action naming**: verbs that match what a user would do (`addProduct(name)`, `filterByCategory(cat)`).
- **Don't put workflows in the page**. If an action requires touching multiple pages, that's a *service* (Layer 7), not a page.
- **Route constants**: if the route isn't in `src/utils/constants.ts` yet, add it under `ROUTES` and import it here. No raw URL strings in the page class.
- **Header reuse**: if the page shows the standard app header, instantiate `HeaderComponent` and expose it as `readonly header: HeaderComponent` so specs can drive cart/menu interactions without leaking selectors.

## Concrete example (built from a hypothetical "Settings" page)

```ts
import { Locator, Page, expect } from '@playwright/test';
import { BasePage } from './base.page';
import { HeaderComponent } from '../components/header.component';
import { ROUTES } from '../utils/constants';

export class SettingsPage extends BasePage {
  readonly header: HeaderComponent;
  readonly title: Locator;
  readonly displayNameInput: Locator;
  readonly emailInput: Locator;
  readonly saveButton: Locator;
  readonly successToast: Locator;

  constructor(page: Page) {
    super(page);
    this.header = new HeaderComponent(page);
    this.title = page.getByRole('heading', { level: 1 });
    this.displayNameInput = page.getByLabel('Display name');
    this.emailInput = page.getByLabel('Email');
    this.saveButton = page.getByRole('button', { name: 'Save changes' });
    this.successToast = page.getByRole('status');
  }

  async goto(): Promise<void> {
    await this.navigate(ROUTES.settings);
    await this.expectLoaded();
  }

  async expectLoaded(): Promise<void> {
    await expect(this.page).toHaveURL(/\/settings/);
    await expect(this.title).toHaveText('Settings');
  }

  async updateDisplayName(name: string): Promise<void> {
    await this.displayNameInput.fill(name);
    await this.saveButton.click();
    await expect(this.successToast).toBeVisible();
  }
}
```
